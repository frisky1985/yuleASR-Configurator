## English

This guide walks you through writing your first yuleASR plugin in about 10
minutes. We'll create a simple code generator that adds a copyright header to
generated C files.

### Prerequisites

- Node.js 18+ and npm
- A yuleASR Configurator monorepo checkout (or just the `@yuletech/plugin-sdk`
  package)
- TypeScript knowledge

### 1. Project Structure

Create a new directory for your plugin:

```
my-plugin/
├── src/
│   └── index.ts          # Plugin entry point
├── package.json          # Package manifest
└── tsconfig.json         # TypeScript config (optional, for local type checking)
```

### 2. Package Configuration

```json
// my-plugin/package.json
{
  "name": "my-copyright-generator",
  "version": "1.0.0",
  "description": "Adds a copyright header to generated C files",
  "main": "dist/index.js",
  "scripts": {
    "build": "tsup src/index.ts --format esm,cjs --dts"
  },
  "dependencies": {
    "@yuletech/plugin-sdk": "^0.1.0"
  },
  "devDependencies": {
    "tsup": "^8.0.0",
    "typescript": "^5.4.0"
  }
}
```

### 3. Writing the Plugin

```typescript
// my-plugin/src/index.ts
import type { YulePlugin, PluginContext } from '@yuletech/plugin-sdk';

const plugin: YulePlugin = {
  id: 'my-copyright-generator',
  name: 'Copyright Header Generator',
  version: '1.0.0',
  type: 'code-generator',
  description: 'Adds a copyright header to generated C files',
  author: 'Your Name',

  async activate(context: PluginContext): Promise<void> {
    context.logger.info('Activating copyright generator plugin');

    context.registerCodeGenerator({
      name: 'CopyrightHeader',
      description: 'Prepends a copyright header to generated C files',
      supportedModules: ['*'], // Supports all modules

      async generate(
        config: Record<string, unknown>,
        options: Record<string, unknown>
      ) {
        // Read user config (set via PUT /v1/api/plugins/:id/config)
        const company = (context.config.company as string) || 'YuleTech';
        const year = new Date().getFullYear();

        const header = `\
/*
 * Copyright (c) ${year} ${company}
 * All rights reserved.
 *
 * This file is part of the yuleASR Configurator project.
 * Generated by ${plugin.name} v${plugin.version}
 *
 * ${'*'.repeat(60)}
 */
`;

        // Wrap the generated file content with the header
        const originalFile = options.originalFile as string | undefined;
        const content = originalFile
          ? `${header}\n${originalFile}`
          : `${header}\n/* Configuration for ${String(config.module || 'unknown')} */\n`;

        return {
          files: [
            {
              path: `${String(config.module || 'default')}_Cfg.h`,
              content,
            },
          ],
        };
      },
    });

    context.logger.info('Copyright generator registered');
  },

  async deactivate(): Promise<void> {
    console.info('[my-copyright-generator] Plugin deactivated');
  },
};

export default plugin;
```

### 4. Building the Plugin

```bash
cd my-plugin
npm install
npm run build
```

Your plugin is now compiled to `dist/index.js` and `dist/index.mjs`.

### 5. Testing with the PluginManager

The simplest way to test is to use the PluginManager directly:

```typescript
import { pluginManager } from '@yuletech/core';
import myPlugin from './my-plugin/dist/index.js';

async function test() {
  // Activate with user config
  const meta = await pluginManager.activate(
    myPlugin,
    { company: 'MyCorp' }, // user config
    'external',
    './dist/index.js'
  );
  console.log('Plugin activated:', meta);

  // List all plugins
  console.log('Plugins:', pluginManager.listPlugins());

  // Deactivate
  await pluginManager.deactivate('my-copyright-generator');
}

test().catch(console.error);
```
